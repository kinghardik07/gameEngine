#include<stdio.h>

int main()
{
	int j=1,k=1,l,m; int n;
	printf("Enter a no. : ");
	scanf("%d",&n);
	printf("\n\n");
	for(int i=1;i<=2*n-1;i++)
	{
		if(i<=n)
		{
			j=1; k=1; l=n-i; m=i;
		}
		
		else
		{
			j=1; k=1; l=i-n; m=2*n-i;
		}

		for(;j<=l;j++)
		{
			printf(" ");
		}		
		
		for(;k<=m;k++)
		{
			printf("*");
		}

	printf("\n");
			
	}
}
